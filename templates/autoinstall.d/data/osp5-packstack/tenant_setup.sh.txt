#!/bin/bash

source ./subr.sh

region={{ region }}

if [ x"$#" != x"1" ]; then
	echo "$0 user"
	exit 1
fi
user=$1; shift

source /root/keystonerc_${user}
nova --os-region-name ${region} keypair-add --pub_key /root/.ssh/id_dsa.pub sshkey
neutron --os-region-name ${region} security-group-rule-create --protocol icmp --direction ingress default
neutron --os-region-name ${region} security-group-rule-create --protocol tcp --port-range-min 22 --port-range-max 22 --direction ingress default
